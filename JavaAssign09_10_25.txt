Q1.Create a Java program where a thread prints numbers from 1 to 10 with a delay of 500 milliseconds between each number

package MultiThreading;

public class NumberPrintingThread implements Runnable {

    @Override
    public void run() {
        for (int i = 1; i <= 10; i++) {
            System.out.println(i);
            try {
                Thread.sleep(500); 
            } catch (InterruptedException e) {
                e.printStackTrace();
            }
        }
    }
}

2.Create a Java program where a thread prints "Hello, World!" 5 times using the Runnable interface.


package MultiThreading;

public class RunnableInterfaceTask implements Runnable{

		private String greeting;
		private int delayTime;
		
		public RunnableInterfaceTask(String greeting, int delayTime) {
			this.greeting = greeting;
			this.delayTime = delayTime;
		}

		@Override
		public void run() {
			for(int a=1;a<=5;a++) {
				System.out.println(greeting);
				try {
					Thread.sleep(delayTime);
				} catch (InterruptedException e) {
					e.printStackTrace();
				}
			}

		}

	}

		t1.start();

	}
}

public class BankAccount {
	    private float balance = 1000;
	    
	    synchronized void deposit (float amount) {
	    System.out.println("Amount Depositing: "+amount);
	    try {
	    Thread.sleep(2000);
	    } catch (InterruptedException e) {
	    
	    e.printStackTrace();
	    }
	    balance =balance+amount;
	    System.out.println("Ammount deposited Successfully");

	    }
   
}
